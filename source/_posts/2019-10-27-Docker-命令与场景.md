# 常用命令

```bash
docker --help
docker-compose --help

# 后台启动当前应用
docker-compose -f docker-compose.yml up -d

# 暂停当前应用
docker-compose -f docker-compose.yml stop

# 关闭容器且删除容器、网络、镜像和卷
docker-compose -f docker-compose.yml kill && docker-compose -f docker-compose.yml down
```
